86  template <class It>
87  It uniquify(It begin, It const end) {
88  std::vector<It> v;
89  v.reserve(static_cast<size_t>(std::distance(begin, end)));
90  for (It i = begin; i != end; ++i) {
91  v.push_back(i);
92  }
93  std::sort(v.begin(), v.end(), target_less());
94  v.erase(std::unique(v.begin(), v.end(), target_equal()), v.end());
95  std::sort(v.begin(), v.end());
96  size_t j = 0;
97  for (It i = begin; i != end && j != v.size(); ++i) {
98  if (i == v[j]) {
99  using std::iter_swap;
100  iter_swap(i, begin);
101  ++j;
102  ++begin;
103  }
104  }
105  return begin;
106  }

116  template <typename T>
117  bool is_power_of_two(const T target) {
118  T mask = 0;
119  T power = static_cast<T>(std::log2l(static_cast<long double>(target)));
120 
121  for (T i = 0; i < power; ++i) {
122  mask += static_cast<T>(1 << i);
123  }
124 
125  return !(target & mask);
126  }
