106  path_iterator(std::string p)
107  : path(p)
108  , crr_pos(0) {
109  ++(*this);
110  }
111 
112  void operator++() {
113  if (crr_pos < path.length())
114  comp = "";
115 
116  while ((crr_pos < path.length()) && (path[crr_pos] != '/') && (path[crr_pos] != '\\')) {
117  comp += path[crr_pos];
118  crr_pos += 1;
119  }
120 
121  crr_pos += 1;
122  }
123 
124  std::string operator*() const {
125  return comp;
126  }
127 
128  operator bool() const {
129  return path.length() >= crr_pos - 1;
130  }
131  };
