010    /**
011     * IAPInfoMIDlet.java
012     * MIDlet to test the APNInfo implementation
013     * 
014     * This class extends the class javax.microedition.midlet.MIDlet. This
015     * class is the entry point in the exampple application. The IAPInfoMIDlet class
016     * creates and maintains references to an IAPTestForm object.
017     * <p>
018     * Note that the IAPInfoMIDlet class has no constructor. MIDlet contructors 
019     * are not required to do anything because intializing of the object is better 
020     * performed in the startApp() method.
021     */
022     
023    public class IAPInfoMIDlet extends MIDlet {
024      
025        /**
026         * This method is called by the runtime
027         * after starting the MIDlet. 
028         * 
029         * Creates the IAPTestForm class, the UI creation
030         * is implemented in the IAPTestForm class
031         * 
032         */
033        public void startApp() {
034            try{
035                IAPTestForm testForm = new IAPTestForm( this );
036            }catch( Throwable e){
037                System.out.println ( "EXCEPTION:" + e.getMessage ()) ;
038                System.err.println( "Exception at startApp method:" + e.toString());
039                e.printStackTrace();
040            }
041        }
042        
043        public void pauseApp() {
044            ;//do nothing
045        }
046        
047        public void destroyApp( boolean unconditional ) {
048            ;//do nothing        
049        }
050    }
